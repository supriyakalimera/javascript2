<script type="text/javascript">
function isPrime(num) {
var count = 0;
for (var i = 2; i < num; i++) {
if (num % i == 0) {
count++;
}
}
if (count == 0) {
return document.write(num + " is a prime number");
} else {
return document.write(num + " is not a prime Number");
}
}
var num = Number(prompt("Enter a Number:"));
isPrime(num);
</script>
