<script type="text/javascript">
var num = Number(prompt("Enter a Number: "));
function getFactorsSum(num) {
var sum = 0;
for (var i = 1; i < num; i++) {
if (num % i == 0) {
sum += i;
}
}
return sum;
}
if (getFactorsSum(num) == num) {
document.write(num + " is a Perfect number");
} else {
document.write(num + " is not a Perfect Number");
}
</script>
