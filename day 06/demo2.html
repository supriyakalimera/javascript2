<script type="text/javascript">
//Printing Fibonacci Sequence
var num = Number(prompt("Enter a Number:"));
var a = 0;
var b = 1;
document.write("Fibonacci Sequence <br/>");
if (num == 1) {
document.write("0");
} else if (num == 2) {
document.write("0 1")
} else if (num > 2) {
document.write(a + " " + b + " ");
while (num > 2) {
var c = a + b;
document.write(c + " ");
a = b;
b = c;
num--;
}
} else {
document.write("Please Enter a Positive Number");
}
</script>
