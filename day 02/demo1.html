<script type="text/JavaScript">
function calculate() {
var num1 = parseInt(document.getElementById("num1").value);
var num2 = parseInt(document.getElementById("num2").value);
var result =
"Sum: " + (num1 + num2) + "<br/>" +
"Sub: " + (num1 - num2) + "<br/>" +
"Mul: " + (num1 * num2) + "<br/>" +
"Quo: " + (num1 / num2) + "<br/>" +
"Rem: " + (num1 % num2) + "<br/>";
document.getElementById('result').innerHTML = result;
}
</script>
